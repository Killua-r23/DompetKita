function renderLaporan(data, tahun) { const bln = data.bulanan; const masuk = Object.values(bln).map(b => b.pemasukan); const keluar = Object.values(bln).map(b => b.pengeluaran); const totalIn = masuk.reduce((a,b) => a+b, 0); const totalOut = keluar.reduce((a,b) => a+b, 0); const labels = ["Jan","Feb","Mar","Apr","Mei","Jun","Jul","Agu","Sep","Okt","Nov","Des"]; document.getElementById("laporanContent").innerHTML = `
Total Pemasukan${fmt(totalIn)}
Total Pengeluaran${fmt(totalOut)}
Net Savings${fmt(totalIn-totalOut)}
Rata-rata Pemasukan/bln${fmt(totalIn/12)}
Rata-rata Pengeluaran/bln${fmt(totalOut/12)}
Rasio Tabungan${totalIn>0 ? Math.round(((totalIn-totalOut)/totalIn)*100) : 0}%
`; setTimeout(() => { const ctx = document.getElementById("chartLaporan").getContext("2d"); if (chartLine) chartLine.destroy(); chartLine = new Chart(ctx, { type: "bar", data: { labels: labels, datasets: [ { label: "Pemasukan", data: masuk, backgroundColor: "rgba(16,185,129,.7)" }, { label: "Pengeluaran", data: keluar, backgroundColor: "rgba(244,63,94,.7)" } ] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { labels: { color: "#94a3b8" } } }, scales: { x: { ticks: { color: "#64748b" } }, y: { ticks: { color: "#64748b", callback: function(v) { if (v >= 1e6) return (v/1e6).toFixed(1) + "jt"; if (v >= 1e3) return (v/1e3).toFixed(0) + "rb"; return v; } } } } } }); }, 100); }
